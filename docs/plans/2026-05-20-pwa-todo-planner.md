# PWA To-do Planner Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the approved single-user, offline-capable Vue PWA planner with Express, SQLite, Docker Compose, daily password unlock, attachments, backup export, and operation-log sync.

**Architecture:** Use a TypeScript npm-workspace monorepo with `apps/web`, `apps/api`, and `packages/shared`. The Vue PWA stores offline state in IndexedDB and syncs operation logs to the Express API, which applies deterministic field-level last-write-wins into SQLite and stores attachments in a Docker volume.

**Tech Stack:** TypeScript, Vue 3, Vite, Pinia, Vue Router, Workbox/Vite PWA, Dexie, Express, better-sqlite3, Zod, Vitest, Vue Test Utils, Supertest, Playwright, Docker Compose.

import { describe, expect, it } from "vitest";
import { canCompleteTask, visibleArchiveItems, sortPlannerItems } from "../src/taskRules";
import type { Task } from "../src/types";

const base = (task: Partial<Task>): Task => ({
  id: task.id ?? "task",
  title: task.title ?? "Task",
  parentId: task.parentId ?? null,
  dueDate: task.dueDate ?? "2026-05-20",
  completedAt: task.completedAt ?? null,
  pinned: task.pinned ?? false,
  tagId: task.tagId ?? null,
  notes: task.notes ?? "",
  recurrence: task.recurrence ?? { type: "none" },
  order: task.order ?? 0,
  createdAt: task.createdAt ?? "2026-05-20T00:00:00.000Z",
  updatedAt: task.updatedAt ?? "2026-05-20T00:00:00.000Z",
  deletedAt: task.deletedAt ?? null
});

describe("task rules", () => {
  it("prevents completing a parent until every child is complete", () => {
    const parent = base({ id: "p" });
    const doneChild = base({ id: "c1", parentId: "p", completedAt: "2026-05-20T01:00:00.000Z" });
    const openChild = base({ id: "c2", parentId: "p" });

    expect(canCompleteTask(parent, [doneChild, openChild])).toBe(false);
    expect(canCompleteTask(parent, [doneChild, { ...openChild, completedAt: "2026-05-20T02:00:00.000Z" }])).toBe(true);
  });

  it("archives completed child tasks only when the parent group is complete", () => {
    const parent = base({ id: "p", completedAt: null });
    const child = base({ id: "c", parentId: "p", completedAt: "2026-05-20T01:00:00.000Z" });
    expect(visibleArchiveItems([parent, child])).toEqual([]);

    const completedParent = { ...parent, completedAt: "2026-05-20T02:00:00.000Z" };
    expect(visibleArchiveItems([completedParent, child]).map((task) => task.id)).toEqual(["p", "c"]);
  });

  it("sorts pinned tasks above unpinned tasks while preserving manual order", () => {
    const tasks = [
      base({ id: "b", pinned: false, order: 1 }),
      base({ id: "a", pinned: true, order: 2 }),
      base({ id: "c", pinned: true, order: 1 })
    ];
    expect(sortPlannerItems(tasks).map((task) => task.id)).toEqual(["c", "a", "b"]);

});
